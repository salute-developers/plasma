```jsx
import React, { useState } from 'react';

import { Global, css } from '@emotion/react';
import { Switch } from '@salutejs/plasma-giga';
import { plasma_giga__light, plasma_giga__dark } from '@salutejs/plasma-themes';

import './style.css';

const lightThemeStyle = css(plasma_giga__light);
const darkThemeStyle = css(plasma_giga__dark);

const App = () => {
    const [theme, setTheme] = useState('light');

    return (
        <>
            <Global styles={theme === 'light' ? lightThemeStyle : darkThemeStyle} />

            <div className="wrapper">
                <Switch
                    label={`app theme: ${theme}`}
                    onChange={() => setTheme((theme) => (theme === 'light' ? 'dark' : 'light'))}
                />
            </div>
        </>
    );
};

export default App;
```
