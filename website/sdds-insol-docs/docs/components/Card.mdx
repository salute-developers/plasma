# Card
Визуальный контейнер, который объединяет различные элементы, такие как изображения, текст и кнопки, в единый блок

<Description name="Card" />
<PropsTable name="Card" />

### Использование

Настройка размера осуществляется явным образом или в зависимости от содержимого, в том числе, за счет указания соотношения сторон.

Настройка внутренних отступов между контентом карточки осуществляется с учетом конкретного контекста использования компонента.

Компонент `Card` является базовым контейнером. `CardContent` используется для вложенного контента внутри `Card`, со своими скруглениями и имеет `overflow: hidden`.
`CardInnerContent` используется для контента вписанного в размеры `CardContent`, имеет `position: absolute`, не влияет на размер всей карточки.

### Пример с вложенным контентом

```tsx live
import React from 'react';
import { Card, CardContent, CardInnerContent, H3, TextS } from '@salutejs/sdds-insol';

export function App() {
    const SvgBackground = (props) => (
        <svg
            xmlns="http://www.w3.org/2000/svg"
            preserveAspectRatio="xMinYMin slice"
            viewBox="1 0.5 8 8"
            {...props}
            style={{ ...props.style, background: 'var(--text-tertiary)' }}
        >
            <path stroke="#fff" strokeDasharray={1} d="M1 1h7M2 2h7M1 3h7M2 4h7M1 5h7M2 6h7M1 7h7M2 8h7" />
        </svg>
    );

    return (
        <Card orientation="horizontal" style={{ width: '250px' }}>
            <CardContent orientation='vertical' aspectRatio='1/1'>
                <CardInnerContent>
                    <div style={{ padding: '30px' }}>
                        <H3>Заголовок</H3>
                        <TextS style={{ paddingTop: '8px' }}>Как оплатить заправку бонусами СберСпасибо?</TextS>
                    </div>
                </CardInnerContent>
                <SvgBackground width="100%" height="100%" style={{ width: '100%', height: '100%' }} />
            </CardContent>
        </Card>
    );
}
```

        import React, { useState } from 'react';
        import { IconPlasma } from '@salutejs/plasma-icons';
        import {
            Button,
            Card,
            CardContent,
            CardInnerContent,
            Flow,
            H1,
            Radiobox,
            RadioGroup,
            shadowDownSoftS,
            Switch,
            TextM,
            TextS,
        } from '@salutejs/sdds-insol';

        export function App() {
            const backgorundColorsMap = {
                card: 'var(--surface-solid-card)',
                primary: 'var(--surface-solid-primary)',
                transparent: 'var(--surface-transparent-card)',
            };

            const contentPositionsMap = {
                initial: 'Контент сверху, картинка снизу',
                reverse: 'Контент снизу, картинка сверху',
            };

            const [hasShadow, setHasShadow] = useState(true);
            const [contentStretching, setContentStretching] = useState(false);
            const [backgroundColor, setBackgroundColor] = useState(backgorundColorsMap.card);
            const [contentPosition, setContentPosition] = useState(contentPositionsMap.initial);

            const getContentPosition = (contentStretching: boolean, contentPosition: string) => {
                if (contentStretching) {
                    return 'spaceBetween';
                }

                switch (contentPosition) {
                    case contentPositionsMap.reverse:
                        return 'end';
                    default:
                        return 'start';
                }
            };

            const Controls = () => {
                return (
                    <Flow orientation="vertical" mainAxisGap="1rem">
                        <Switch
                            size="s"
                            toggleSize="s"
                            label="Тень"
                            checked={hasShadow}
                            onChange={(event) => {
                                setHasShadow(event.target.checked);
                            }}
                        />
                        <Switch
                            size="s"
                            toggleSize="s"
                            label="Растянуть контент"
                            checked={contentStretching}
                            onChange={(event) => {
                                setContentStretching(event.target.checked);
                            }}
                        />
                        <RadioGroup aria-labelledby="radiogroup-title-id">
                            <TextS id="radiogroup-title-id" style={{ marginBottom: '0.5rem' }}>
                                Цвет подложки карточки
                            </TextS>
                            {Object.entries(backgorundColorsMap).map(([key, value]) => (
                                <Radiobox
                                    key={key}
                                    name={key}
                                    value={key}
                                    label={key}
                                    checked={backgroundColor === value}
                                    size="s"
                                    onChange={() => {
                                        setBackgroundColor(backgorundColorsMap[key]);
                                    }}
                                />
                            ))}
                        </RadioGroup>

                        <RadioGroup aria-labelledby="radiogroup-content-title-id">
                            <TextS id="radiogroup-content-title-id" style={{ marginBottom: '0.5rem' }}>
                                Положение контента на карточке
                            </TextS>
                            {Object.entries(contentPositionsMap).map(([key, value]) => (
                                <Radiobox
                                    key={key}
                                    name={key}
                                    value={key}
                                    label={value}
                                    checked={contentPosition === value}
                                    size="s"
                                    onChange={() => {
                                        setContentPosition(contentPositionsMap[key]);
                                    }}
                                />
                            ))}
                        </RadioGroup>
                    </Flow>
                );
            };

            const SvgBackground = (props) => (
                <svg
                    xmlns="http://www.w3.org/2000/svg"
                    preserveAspectRatio="xMinYMin slice"
                    viewBox="1 0.5 8 8"
                    {...props}
                    style={{ ...props.style, background: 'var(--text-tertiary)' }}
                >
                    <path stroke="#fff" strokeDasharray={1} d="M1 1h7M2 2h7M1 3h7M2 4h7M1 5h7M2 6h7M1 7h7M2 8h7" />
                </svg>
            );

            const TextBox = () => {
                return (
                    <Flow orientation="vertical" mainAxisGap="1rem" style={{ marginBottom: '1.5rem' }}>
                        <IconPlasma color="var(--text-accent)" />
                        <H1 bold>Title</H1>
                        <TextM>SubTitle</TextM>
                    </Flow>
                );
            };

            return (
                <div>
                    <Flow orientation="vertical" mainAxisGap="3rem">
                        <Controls />
                        <Card
                            orientation="vertical"
                            backgroundType="solid"
                            style={{
                                width: '37.5rem',
                                height: '37.5rem',
                                padding: 0,
                                borderRadius: '2rem',
                                overflow: 'hidden',
                                backgroundColor,
                                ...(hasShadow && { boxShadow: shadowDownSoftS }),
                            }}
                        >
                            <CardContent orientation="vertical" aspectRatio="1/1" style={{ padding: '2.75rem 3rem' }}>
                                <Flow
                                    orientation="vertical"
                                    arrangement={getContentPosition(contentStretching, contentPosition)}
                                    style={{ width: '100%', height: '100%' }}
                                >
                                    <TextBox />
                                    <Button text="label" />
                                </Flow>
                            </CardContent>
                            <CardInnerContent
                                orientation="vertical"
                                style={{
                                    justifyContent: contentPosition === contentPositionsMap.initial ? 'end' : 'start',
                                    alignItems: 'end',
                                    zIndex: '-1',
                                }}
                            >
                                <SvgBackground style={{ width: '15rem', height: '15rem', borderRadius: '0.375rem' }} />
                            </CardInnerContent>
                        </Card>
                    </Flow>
                </div>
            );
        }
