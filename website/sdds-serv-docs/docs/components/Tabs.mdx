# Tabs

Структура для вкладок похожа на структуру маркированных списков.

## Использование

### Стандартное подключение, без клавиатурной навигации

```tsx live
import React, { useState } from 'react';
import { Tabs, TabItem } from '@salutejs/sdds-serv';
import { IconClock } from '@salutejs/plasma-icons';

export function App() {
    const items = Array(4).fill(0);
    const [index, setIndex] = useState(0);

    return (
        <div>
            <Tabs view="filled" stretch size="xs">
                {items.map((_, i) => (
                    <TabItem
                        view="secondary"
                        key={`item:${i}`}
                        size="xs"
                        selected={i === index}
                        tabIndex={0}
                        contentLeft={<IconClock size="xs" color="inherit" />}
                        onClick={() => setIndex(i)}
                    >
                        {`Label${i + 1}`}
                    </TabItem>
                ))}
            </Tabs>
        </div>
    );
}
```

### Расположение табов
Табы могут быть горизонтальными (по умолчанию) и вертикальными. За это отвечает свойство `orientation`.

```tsx live
import React, { useState } from 'react';
import { Tabs, TabItem, IconTabItem } from '@salutejs/sdds-serv';
import { IconClock } from '@salutejs/plasma-icons';

export function App() {
    const items = Array(8).fill(0);
    const [index, setIndex] = useState(0);

    return (
        <div style={{ gap: '2rem' }}>
            <Tabs view="divider" orientation="vertical" size="xs" style={{ width: '15rem' }}>
                {items.map((_, i) => (
                    <TabItem
                        view="divider"
                        orientation="vertical"
                        key={`item:${i}`}
                        size="xs"
                        selected={i === index}
                        tabIndex={0}
                        contentLeft={<IconClock size="xs" color="inherit" />}
                        onClick={() => setIndex(i)}
                    >
                        {`Label${i + 1}`}
                    </TabItem>
                ))}
            </Tabs>

            <Tabs view="divider" orientation="vertical" size="xs" style={{ width: '15rem' }}>
                {items.map((_, i) => (
                    <IconTabItem
                        view="divider"
                        orientation="vertical"
                        key={`item:${i}`}
                        size="xs"
                        selected={i === index}
                        tabIndex={0}
                        onClick={() => setIndex(i)}
                    >
                        <IconClock size="xs" color="inherit" />
                    </IconTabItem>
                ))}
            </Tabs>
        </div>
    );
}
```

### Пример с прокруткой
При выборе Tab происходит прокрутка до выбранного TabItem.

```tsx live
import React, { useState } from 'react';
import { Tabs, TabItem, IconTabItem, Button } from '@salutejs/sdds-serv';
import { IconClock } from '@salutejs/plasma-icons';

export function App() {
    const items = Array(8).fill(0);
    const [index, setIndex] = useState(0);

    const specificIndex = 6;

    return (
        <div style={{ display:'flex', flexDirection: 'column', gap: '1rem' }}>
            <Tabs view="divider" size="xs" style={{ width: '15rem' }}>
                {items.map((_, i) => (
                    <TabItem
                        view="divider"
                        key={`item:${i}`}
                        size="xs"
                        selected={i === index}
                        tabIndex={0}
                        contentLeft={<IconClock size="xs" color="inherit" />}
                        onClick={() => setIndex(i)}
                    >
                        {`Label${i + 1}`}
                    </TabItem>
                ))}
            </Tabs>

            <Tabs view="divider" size="xs" style={{ width: '15rem' }}>
                {items.map((_, i) => (
                    <TabItem
                        view="divider"
                        key={`item:${i}`}
                        size="xs"
                        selected={i === index}
                        tabIndex={0}
                        onClick={() => setIndex(i)}
                    >
                        <IconClock size="xs" color="inherit" />
                    </TabItem>
                ))}
            </Tabs>

            <Button onClick={() => setIndex(specificIndex)}>{`Выбрать элемент ${specificIndex + 1}`}</Button>
        </div>
    );
}
```

### Пример с Dropdown

```tsx live
import React, { useState } from 'react';
import { Tabs, TabItem, Dropdown } from '@salutejs/sdds-serv';
import { IconClock } from '@salutejs/plasma-icons';

export function App() {
    const items = Array(8).fill(0);
    const [index, setIndex] = useState(0);

    const maxItemQuantity = 3;
    const visibleItems = items.slice(0, maxItemQuantity);
    const otherItems = items.slice(maxItemQuantity);

    const dropdownItems = otherItems.map((_, i) => {
        const itemIndex = maxItemQuantity + i;

        return {
            label: `Label${itemIndex + 1}`,
            value: itemIndex,
        };
    });

    return (
        <div style={{ display:'flex', flexDirection: 'column', gap: '1rem' }}>
            <div style={{ height: '15rem', alignItems: 'flex-start' }}>
                <Tabs clip="showAll" view="divider" size="xs">
                    {visibleItems.map((_, i) => (
                        <TabItem
                            key={`item:${i}`}
                            view="divider"
                            selected={i === index}
                            onClick={() => setIndex(i)}
                            tabIndex={0}
                            size="xs"
                        >
                            {`Label${i + 1}`}
                        </TabItem>
                    ))}
                    {dropdownItems.length > 0 && (
                        <div style={{ marginLeft: '1.75rem' }}>
                            <Dropdown size="xs" items={dropdownItems} onItemSelect={(item) => setIndex(item.value)}>
                                <TabItem key="item:ShowAll" view="divider" tabIndex={0} size="xs">
                                    ShowAll
                                </TabItem>
                            </Dropdown>
                        </div>
                    )}
                </Tabs>
            </div>

            <div style={{ height: '15rem', alignItems: 'flex-start' }}>
                <Tabs clip="showAll" view="divider" size="xs">
                    {visibleItems.map((_, i) => (
                        <IconTabItem
                            key={`item:${i}`}
                            view="divider"
                            selected={i === index}
                            onClick={() => setIndex(i)}
                            tabIndex={0}
                            size="xs"
                        >
                            <IconDotsHorizontalOutline color="inherit" size="xs" />
                        </IconTabItem>
                    ))}
                    {dropdownItems.length > 0 && (
                        <div style={{ marginLeft: '1.75rem' }}>
                            <Dropdown size="xs" items={dropdownItems} onItemSelect={(item) => setIndex(item.value)}>
                                <IconTabItem key="item:ShowAll" view="divider" tabIndex={0} size="xs">
                                    <IconClock color="inherit" size="xs" />
                                </IconTabItem>
                            </Dropdown>
                        </div>
                    )}
                </Tabs>
            </div>
        </div>
    );
}
```

### Слот дополнительного действия в `TabItem`
Для задания дополнительного действия используется свойство `actionContent`

```tsx live

import { Tabs, TabItem } from '@salutejs/sdds-serv';
import { IconClose } from '@salutejs/plasma-icons';

export function App() {
    const items = Array(4).fill(0);
    const [index, setIndex] = useState(0);

    return (
        <div>
            <Tabs view="filled" stretch size="xs">
                {items.map((_, i) => (
                    <TabItem
                        view="secondary"
                        key={`item:${i}`}
                        size="xs"
                        selected={i === index}
                        tabIndex={0}
                        actionContent={<IconClose size="xs" color="inherit" />}
                        onClick={() => setIndex(i)}
                    >
                        {`Label${i + 1}`}
                    </TabItem>
                ))}
            </Tabs>
        </div>
    );
}
```

### Подключение клавиатурной навигации
Для этого необходимо дополнительно прокинуть свойства `index, itemIndex, onIndexChange`.<br/>
Для горизонтальных табов: клавиши ArrowLeft, ArrowRight, Home, End для навигации по вкладкам.<br/>
Для вертикальных табов: клавиши ArrowUp, ArrowDown, Home, End для навигации по вкладкам.

```tsx live
import React, { useState } from 'react';
import { Tabs, TabItem } from '@salutejs/sdds-serv';
import { IconClock } from '@salutejs/plasma-icons';

export function App() {
    const items = Array(4).fill(0);
    const [horizontalIndex, setHorizontalIndex] = useState(0);
    const [verticalIndex, setVerticalIndex] = useState(0);

    return (
        <div style={{ display: "flex", flexDirection: "column", gap: "50px"}}>
            <div>
                <Tabs view="divider" stretch size="xs" index={horizontalIndex}>
                    {items.map((_, i) => (
                        <TabItem
                            view="divider"
                            key={`item:${i}`}
                            size="xs"
                            itemIndex={i}
                            onIndexChange={(i) => setHorizontalIndex(i)}
                            selected={i === horizontalIndex}
                            tabIndex={0}
                            contentLeft={<IconClock size="xs" color="inherit" />}
                            onClick={() => setHorizontalIndex(i)}
                        >
                            {`Label${i + 1}`}
                        </TabItem>
                    ))}
                </Tabs>
            </div>

            <div>
                <Tabs orientation="vertical" view="divider" size="xs" index={verticalIndex}>
                    {items.map((_, i) => (
                        <TabItem
                            orientation="vertical"
                            view="divider"
                            key={`item:${i}`}
                            size="xs"
                            itemIndex={i}
                            onIndexChange={(i) => setVerticalIndex(i)}
                            selected={i === verticalIndex}
                            tabIndex={0}
                            contentLeft={<IconClock size="xs" color="inherit" />}
                            onClick={() => setVerticalIndex(i)}
                        >
                            {`Label${i + 1}`}
                        </TabItem>
                    ))}
                </Tabs>
            </div>
        </div>
    );
}
```
