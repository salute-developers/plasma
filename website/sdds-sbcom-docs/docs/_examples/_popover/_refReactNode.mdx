```tsx live
import React, { useState } from 'react';
import styled from "styled-components";
import { Popover, Button, H4, ButtonGroup } from '@salutejs/sdds-sbcom';

export function App() {
    const [isOpen, setIsOpen] = useState(false);

    const StyledContent = styled.div`
        display: flex;
        flex-direction: column;
        align-items: center;

        padding: 1rem;

        background-color: gray;
    `;

    return (
        <section style={{ display: 'flex', width: "100%", "justify-content": 'center', height: '16rem' }}>
            <Popover
                opened={isOpen}
                onToggle={(is) => setIsOpen(is)}
                role="presentation"
                id="popover"
                target={<Button>Target</Button>}
                offset={[0, 6]}
                placement='bottom'
                trigger='click'
                hasArrow
                closeOnOverlayClick
                closeOnEsc
                isFocusTrapped
            >
                <StyledContent>
                    <H4>Заголовок H4</H4>
                    <ButtonGroup style={{ "justify-content": 'flex-end', paddingTop: '24px' }} isCommonButtonStyles={false} gap="wide">
                        <Button view="accentFiled" onClick={() => setIsOpen(false)}>Принять</Button>
                        <Button onClick={() => setIsOpen(false)}>Закрыть</Button>
                    </ButtonGroup>
                </StyledContent>
            </Popover>
        </section>
    );
}
```
