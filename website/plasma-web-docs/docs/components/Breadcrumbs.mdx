# Обычное использование

```tsx live
import React from 'react';
import { Breadcrumbs } from '@salutejs/plasma-web';

export function App() {
    const items = [
        { title: 'Home', href: '/' },
        { title: 'About as', href: '/' },
        { renderItem: () => <span>Custom Item</span> },
        { title: 'Contacts' },
    ];

    return (
        <div>
            <Breadcrumbs items={items} />
        </div>
    );
}
```
# Использование с укоротителем
```tsx live
import React from 'react';
import { Breadcrumbs } from '@salutejs/plasma-web';

export function App() {
    const items = [
        { title: 'Home', href: '/' },
        { title: 'About as', href: '/' },
        { renderItem: () => <span>Custom Item</span> },
        { title: 'Contacts' },
    ];

    return (
        <div style={{"flexDirection": 'column'}}>
            <div>
                <Breadcrumbs items={items} showItems={2} />
            </div>
            <div>
                 <Breadcrumbs items={items} showItems={3} />
            </div>
        </div>
    );
}
```

# Использование с кастомным элементом
```tsx live
import React from 'react';
import { Breadcrumbs, Dropdown } from '@salutejs/plasma-web';

export function App() {
    const items = [
        { title: 'Home', href: '/' },
        { title: 'About as', href: '/' },
        {
            renderItem: () => {
                const itemsDropdown = [
                    {
                        value: 'Custom Item 1',
                        label: 'Custom Item 1',
                    },
                    {
                        value: 'Custom Item 2',
                        label: 'Custom Item 2',
                    },
                ];
                return (
                    <Dropdown variant="tight" placement="bottom" items={itemsDropdown} trigger="hover">
                        <span>...</span>
                    </Dropdown>
                );
            },
        },
        { title: 'Contacts' },
    ];


    return (
        <div style={{height: '200px'}}>
            <Breadcrumbs items={items} />
        </div>
    );
}
```
